void LCRS_DestroyTree(LCRSNode* Root){
	if(Root->RightSibling != NULL)
		LCRS_DestoryTree(Root->RightSibling);
	
	if(Root->LeftChild != NULL)
		LCRS_DestroyTree(Root->LeftChild);

	Root->LeftChild = NULL;
	Root->RightSibling = NULL;

	LCRS_DestoryNode(Root);
